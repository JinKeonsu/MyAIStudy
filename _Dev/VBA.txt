// 1) option price & mgmt.

Sub btn_click()
  Dim nRows as integer

  nRows = Sheet1.Cells(Rows.Count, 1).End(xlUp).Row

  For i=2 to nRows
    strValue = Sheet1.Cells(i,2).Value
    Dim arrOption() as string
    Dim strPrice as string
    Dim strMgmt as string

    strPrice = ""
    strMgmt = ""
    arrOption = Split(strValue, chr(10))

    For j=0 to UBound(arrOption)
      strPrice = strPrice & "0"
      strOne = "N," & Sheet1.Cells(i,1).Value & "," & arrOption(j)
      strMgmt = strMgmt & strOne

      if j<UBound(arrOption) then
        strPrice = strPrice & chr(10)
        strMgmt = strMgmt & chr(10)
      end if
    Next

    Sheet1.Cells(i, 3).Value = strPrice
    Sheet1.Cells(i, 4).Value = strMgmt

  Next

End Sub
